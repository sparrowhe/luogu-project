不高兴的津津

简单的数学计算

#include<iostream>
using namespace std;
int main()
{
    int a,b,sum,max=0,day=0,i;
    for(i=1;i<8;i++)
    {
        cin>>a>>b;
        sum=a+b;
        if((sum>8) && (sum>max)) max=sum,day=i;
    }
    cout<<day; 
    return 0;
}
